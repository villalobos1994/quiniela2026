/*********************** 🧊 BOMBOS ************************/ let bombo1 = []; let bombo2 = []; let bombo3 = []; let bombo4 = []; let grupos = []; let tabla = {}; /*********************** 🎰 SHUFFLE (SORTEO REAL) ************************/ function shuffle(array) { for (let i = array.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [array[i], array[j]] = [array[j], array[i]]; } } /*********************** 🏆 GENERAR GRUPOS ************************/ function generarGrupos() { shuffle(bombo1); shuffle(bombo2); shuffle(bombo3); shuffle(bombo4); grupos = []; for (let i = 0; i < bombo1.length; i++) { grupos.push({ nombre: `Grupo ${String.fromCharCode(65 + i)}`, equipos: [ bombo1[i], bombo2[i], bombo3[i], bombo4[i] ] }); } renderGrupos(); } /*********************** 🖥️ MOSTRAR GRUPOS (UI) ************************/ function renderGrupos() { console.clear(); console.log("===== GRUPOS ====="); grupos.forEach(g => { console.log(`\n${g.nombre}`); g.equipos.forEach(e => console.log("- " + e)); }); } /*********************** ⚽ INICIALIZAR TABLA ************************/ function inicializarTabla() { let todos = [...bombo1, ...bombo2, ...bombo3, ...bombo4]; todos.forEach(eq => { tabla[eq] = { puntos: 0, jugados: 0, ganados: 0, empatados: 0, perdidos: 0 }; }); } /*********************** 📊 RESULTADOS ************************/ function registrarResultado(equipoA, equipoB, golesA, golesB) { tabla[equipoA].jugados++; tabla[equipoB].jugados++; if (golesA > golesB) { tabla[equipoA].ganados++; tabla[equipoA].puntos += 3; tabla[equipoB].perdidos++; } else if (golesB > golesA) { tabla[equipoB].ganados++; tabla[equipoB].puntos += 3; tabla[equipoA].perdidos++; } else { tabla[equipoA].empatados++; tabla[equipoB].empatados++; tabla[equipoA].puntos += 1; tabla[equipoB].puntos += 1; } } /*********************** 🧾 MOSTRAR TABLA ************************/ function mostrarTabla() { let ranking = Object.entries(tabla) .sort((a, b) => b[1].puntos - a[1].puntos); console.clear(); console.log("===== TABLA ====="); ranking.forEach(([eq, d]) => { console.log(`${eq} | ${d.puntos} pts | PJ:${d.jugados} G:${d.ganados} E:${d.empatados} P:${d.perdidos}`); }); } /*********************** 🔥 COMPATIBILIDAD CON TU HTML VIEJO (ESTO ARREGLA TU GITHUB PAGES) ************************/ function mostrar() { renderGrupos(); } function sortear() { generarGrupos(); } function iniciar() { inicializarTabla(); generarGrupos(); mostrarTabla(); }
